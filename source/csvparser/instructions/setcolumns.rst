   use Cajudev\CsvParser;

   $dir = __DIR__ . '/file.csv';

   $csv = new CsvParser($dir);

   $csv->setDelimiter(';');

   $csv->setColumns(['Produto', 'Preço', 'Pagamento', 'Nome']); // <<< Linha acrescentada

   $results = $csv->parse();

   print_r($results);
